CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    email VARCHAR(255),
    gender VARCHAR(10),
    ip_address VARCHAR(15));

CREATE TABLE users_statistic (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    date VARCHAR(10),
    page_views INTEGER,
    clicks INTEGER);
